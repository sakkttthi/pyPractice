Install python & pip
    python --version
    pip --version

Install selenium
    pip install selenium
    pip uninsall selenium

Install Robot frameworl
    pip install robotframework
    pip show robotframework

    pip install --upgrade robotframework # to upgrade
    pip install robotframework==2.9.2 # to get specific version

    pip uninstall robotframework

    pip list #package information
    pip check robotframework

Install robotframework selenium library
    pip install robotframework-seleniumlibrary
    pip uninstall robotframework-seleniumlibrary
